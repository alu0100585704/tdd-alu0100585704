def insertHead(value)
    nodo = Nodo.new(value,nil,nil)
    nodo.next = @head
    nodo.prev = nil
    @head.prev = nodo  if !@head.nil?
    @head = nodo        
    @tail = nodo if @tail.nil?
        
    
end
